sudo apt install libguestfs-tools

cp /var/lib/libvirt/images/ubuntu24.04.qcow2  /var/lib/libvirt/images/ubuntu1.qcow2

sudo virt-sysprep -a /var/lib/libvirt/images/ubuntu1.qcow2 --hostname ubuntu1 --root-password password:asd --enable customize,dhcp-client-state,net-hostname,net-hwaddr,machine-id

sudo virt-customize -a /var/lib/libvirt/images/ubuntu1.qcow2 --password root:password:asd

sudo virt-customize -a /var/lib/libvirt/images/ubuntu1.qcow2  --ssh-inject ubuntu:file:/root/.ssh/id_rsa.pub

sudo virt-install --name ubuntu1 --ram 2048 --vcpus 2 --disk path=/var/lib/libvirt/images/ubuntu1.qcow2 --import --os-variant ubuntu24.04 --network default --graphics vnc --noautoconsole


sudo virsh domifaddr <hostname z.B. ubuntu1>
